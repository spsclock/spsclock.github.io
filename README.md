function startTime() {
    var today = new Date();
    var h = today.getHours();
    var m = today.getMinutes();
    var s = today.getSeconds();
    var weekday = today.getDay();
    var month = today.getMonth();
    var d = today.getDate();

    var weekdays=
["Sun","Mon","Tue","Wed","Thu","Fri","Sat"];

    var months=
["Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"];

    var period;
    var remaining;
    var elapsed;
    var dayminute=h*60+m;

var periodends=[
08*60+55, //ie 8:55am, in minutes from start of day
09*60+50, //ie 9:50am, in minutes from start of day
10*60+40,
11*60+05, //end of short break
11*60+55,
12*60+50,
14*60+30, //end of lunch
15*60+20,
16*60+15,
];

var periodnames=["TU","p1","p2","SB","p3","p4","LT","p5","p6"];

for (i = 8; i >= 0; i--) 
{
   if (dayminute<periodends[i]) 
       {period=periodnames[i];
        remaining=checkTime(periodends[i]-dayminute);
       }
}

if (h>12) {h-= 12}; //display as 12-hour clock
    h = checkTime(h);
    m = checkTime(m);
    s = checkTime(s);

if (remaining>50) {remaining='('+remaining+')';}

output = weekdays[weekday]+" "+d+" "+months[month];
output += "<br> "+h + ":" + m + ":" + s;
output += " <br>"+period+" left:"+  remaining;

if (remaining<51 && period != "LT" && period != "SB" && period != "TU") {
elapsed=50-remaining;
output += "<br>elapsed:"+ elapsed;
				}


    document.getElementById('txt').innerHTML = output
    
    var t = setTimeout(startTime, 500);
}

    if (i < 10) {i = "0" + i};  // add zero in front of numbers < 10
    return i;
